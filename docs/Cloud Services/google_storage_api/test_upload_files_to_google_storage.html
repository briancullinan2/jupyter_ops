var streamToGoogle = importer.import("upload files google cloud");

    it('should upload a file to a bucket', () => {
        return streamToGoogle(

            'sheet-to-web.com')
            .then(url => {
                assert(url.length > 0, 'should have a file url');
            })
    })
    
    
})
