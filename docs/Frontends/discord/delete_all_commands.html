async function deleteCommands(guildId) {
  var toRemove = await getCommands(guildId)
  for(var i = 0; i < toRemove.length; i++) {
    if(EXCEPT_COMMANDS.includes(toRemove[i].name))
      continue
    await timeout(3000)
    await deleteCommand(toRemove[i].id, guildId)
  }
}
