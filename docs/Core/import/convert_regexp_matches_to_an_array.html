The regexToArray function takes a regular expression object, a string to search, and an optional parameter to customize the return value. It returns an array of matches based on the provided parameters, such as the start position and length of each match, all match groups, or a specific match group.

function regexToArray(ex, str, i = 0) {
    var co = []
    var m
    while ((m = ex.exec(str))) {
        co.push(i === -1 ? [m.index, m[0].length] : i === false ? m : m[i])
    }
    return co
}

module.exports.regexToArray = regexToArray;

Function: regexToArray

Parameters

  • ex: A regular expression object
  • str: The string to search for matches
  • i: An optional parameter, default value is 0. Can be one of the following:
    • -1: Return the start position and length of each match
    • false: Return all match groups
    • index of match group: Return only the match group at the specified index

Return Value

An array of matches based on the provided parameters

Usage

const regexToArray = require('./regexToArray');

const regex = /a+/g;
const str = 'aaabbbcc';
const result1 = regexToArray(regex, str, -1);
console.log(result1); // Output: [[0, 3], [4, 3], [8, 2]]

const result2 = regexToArray(regex, str, false);
console.log(result2); // Output: ['aaa', 'bbb', 'cc']

const result3 = regexToArray(regex, str, 0);
console.log(result3); // Output: ['aaa']
