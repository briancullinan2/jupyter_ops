function processMetaKernel(meta_kernel) {
    var meta = interface(meta_kernel, metaKernelInterface);
    var kernel = extend(meta, {
        do_init,
        do_shutdown,
        do_message,
        do_execute,
        do_complete,
        do_inspect,
        do_history,
        do_is_complete,
    })
    return socketMetaKernel(kernel);
}
