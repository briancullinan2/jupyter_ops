// Use define function to define a function signature
const res = define({
  sum: {
      library: "libsum",
      retType: DataType.I32,
      paramsType: [DataType.I32, DataType.I32],
  },
  atoi: {
      library: "libnative",
      retType: DataType.I32,
      paramsType: [DataType.String],
  }
})
equal(res.sum([1, 2]), 3)
equal(res.atoi(["1000"]), 1000)
