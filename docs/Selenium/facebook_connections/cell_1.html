var importer = require('../Core');
var runSeleniumCell = importer.import('selenium cell');

$$.async();
var loginFacebook, likeAllPosts;
runSeleniumCell([
    'log in facebook',
    'like all facebook posts',
])
    .then(r => {
        ({
            loginFacebook,
            likeAllPosts,
        } = r);
        return likeAllPosts('https://www.facebook.com/megamindbc', true);
    })
    .then(r => $$.sendResult(r))
    .catch(e => $$.sendError(e))
