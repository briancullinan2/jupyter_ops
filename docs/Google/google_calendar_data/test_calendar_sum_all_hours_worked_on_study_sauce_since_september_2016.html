// Define a function to get events by date range and query
async function getEventsByDateRangeAndQuery(timeMin, timeMax, q) {
  try {
    const response = await listEvents({
      timeMin,
      timeMax,
      q
    });
    return response.data.items;
  } catch (error) {
    // TODO: Handle API errors
    console.error(error);
    return [];
  }
}

// Define a function to sum events
async function sumEventsByDateRangeAndQuery(timeMin, timeMax, q) {
  const events = await getEventsByDateRangeAndQuery(timeMin, timeMax, q);
  const sum = events.reduce((acc, event) => acc + event.duration, 0);
  return sum;
}

// Define a function to convert date to ISO string
function dateToISO(date) {
  return new Date(date).toISOString();
}

// Example usage
async function main() {
  const timeMin = dateToISO('2016/09/22');
  const timeMax = dateToISO('2017/09/22');

    const sum = await sumEventsByDateRangeAndQuery(timeMin, timeMax, q);
