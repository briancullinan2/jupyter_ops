This code retrieves Google Calendar events matching a specific query ("study sauce") between two dates and then summarizes those events, sending the result to a client.

var importer = require('../Core');
var { sumEvents, listEvents, ISODateString } = importer.import([
    'import google calendar api',
    'how to list events',
    'sum a list of events',
    'convert date to ISO'
]);

$$.async();
listEvents({
    timeMin: ISODateString(new Date('2016/09/22')),
    timeMax: ISODateString(new Date('2017/09/22')),
    q: 'study sauce'
})
    .then(r => $$.sendResult(sumEvents(r)))
    .catch(e => $$.sendError(e));
