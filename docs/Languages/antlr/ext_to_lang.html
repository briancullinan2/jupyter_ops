function extToLang(filename) {
    var result = ext_to_lang[filename]
        || filename.replace(/^\./, '')
        || ext_to_lang[path.extname(filename).replace(/^\./, '')]
    if(!result) throw new Error(`Unknown filetype ${filename}`)
    return result
}

export default extToLang
