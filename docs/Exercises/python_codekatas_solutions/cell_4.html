def simplify_path(path):
    """Simplify Unix-style file path."""
    stack = []
    tokens = [t for t in path.split('/') if t != '.' and t != '']
    for token in tokens:
        if token != '..':
            stack.append(token)
        elif stack:
            stack.pop()
    return '/'+'/'.join(stack)

assert simplify_path('/') == '/'
assert simplify_path('/../') == '/'
assert simplify_path('/...') == '/...'
assert simplify_path('/.../') == '/...'
assert simplify_path('/foo/..') == '/'
assert simplify_path('/foo///.//bar//') == '/foo/bar'
print('All passed!')
