class ListNode(object):
    def __init__(self, x):
        self.val = x
        self.next = None
        
    def __str__(self): # for your debugging purposes
        return str(self.val) + '->' + str(self.next)
    
    def __eq__(self, other): # for asserts to work
            return (isinstance(other, self.__class__) 
                and self.__dict__ == other.__dict__)
        
def reverse_list(head, prev=None):
    """Recursive solution."""
    if not head:
        return prev
    nxt = head.next
    head.next = prev
    return reverse_list(nxt, head)

head = ListNode(1)
rev = ListNode(1)
assert reverse_list(head) == head

  • prev: The previous node in the reversed list (default value is None).
