function isFib(num) {
    let last = 0;
    let next = 1;
    while (next <= num) {
        let newThingThatIsAVariable = last + next;
        if (newThingThatIsAVariable === num) { return true; }
        last = next;
        next = newThingThatIsAVariable;
    }
    return false;
}

isFib(4);

// 1 + 1 = 2
