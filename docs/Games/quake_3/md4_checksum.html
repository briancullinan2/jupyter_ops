function copy64(M, input)
{
    for (var i=0;i<16;i++)
        M[i] =
            (input[i*4+3] << 24) |
            (input[i*4+2] << 16) |
            (input[i*4+1] << 8) |
            (input[i*4+0] << 0)
}

function mdfour_update(md, input, n)
{
    var M = new Uint32Array(16)

    m = md

    if (n == 0) mdfour_tail(input, n)
